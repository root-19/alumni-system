# QUICK FIX COMMANDS - Copy-paste lang sa Laravel Cloud SSH

# Step 1: Remove old symlink
rm -f public/storage
rm -rf public/storage

# Step 2: Create directories
mkdir -p storage/app/public/news_images
mkdir -p storage/app/public/alumni-posts

# Step 3: Create symlink
php artisan storage:link

# Step 4: If artisan fails, create manually
cd public && ln -s ../storage/app/public storage && cd ..

# Step 5: Set permissions
chmod -R 775 storage/app/public
chmod 755 public

# Step 6: Verify
ls -la public/storage

# Step 7: Clear caches
php artisan view:clear
php artisan config:clear

# Step 8: Test
ls public/storage/alumni-posts/
